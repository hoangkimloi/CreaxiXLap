CREATE DATABASE chat_servers;
USE chat_servers;

CREATE TABLE ACCOUNT
(
  ID_User INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
  User_Name varchar(50) NOT NULL
);
CREATE TABLE MESSAGE
(
   ID_MESS INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
   MESSAGE TEXT,
   ID_User INT,
   ID_Senter INT,
   ID_Receive INT,
   ID_Timesent INT,
   FOREIGN KEY(ID_User) REFERENCES ACCOUNT(ID_User)
);
CREATE TABLE FRIEND
(
  ID_Relation INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
  ID_User1 INT,
  ID_User2 INT,
  FOREIGN KEY(ID_User1) REFERENCES ACCOUNT(ID_User),
  FOREIGN KEY(ID_User2) REFERENCES ACCOUNT(ID_User)
);
